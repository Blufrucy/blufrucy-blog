<div class="blog-filter" id="blog-filter-zh">
  <span class="blog-filter-label">按标签筛选：</span>
  <button class="blog-filter-pill active" data-tag="all">全部</button>
</div>

<ul class="blog-list" id="blog-list-zh"></ul>

</div>

<script>
;(function () {
  // Wait for DOM + Quartz SPA
  function init() {
    const enList = document.getElementById("blog-list-en");
    const zhList = document.getElementById("blog-list-zh");
    if (!enList || !zhList) return;

    fetch("/static/contentIndex.json")
      .then((r) => r.json())
      .then((index) => {
        // Filter docs/ pages, exclude virtual folder index
        const posts = Object.values(index).filter(
          (p) => p.slug && p.slug.startsWith("docs/") && p.slug !== "docs/index"
        );

        if (posts.length === 0) {
          enList.innerHTML = '<li class="blog-empty">No posts yet.</li>';
          zhList.innerHTML = '<li class="blog-empty">暂无文章。</li>';
          return;
        }

        // Collect all unique tags
        const tagSet = new Set();
        posts.forEach((p) => (p.tags || []).forEach((t) => tagSet.add(t)));
        const allTags = Array.from(tagSet).sort();

        // Build filter pills
        buildFilterPills("blog-filter-en", allTags, enList, posts);
        buildFilterPills("blog-filter-zh", allTags, zhList, posts);

        // Render
        renderPosts(enList, posts, "en");
        renderPosts(zhList, posts, "zh");
      })
      .catch(() => {
        enList.innerHTML = '<li class="blog-empty">Failed to load posts.</li>';
        zhList.innerHTML = '<li class="blog-empty">加载文章失败。</li>';
      });
  }

  function buildFilterPills(filterId, allTags, listEl, posts) {
    const filter = document.getElementById(filterId);
    if (!filter) return;
    // Keep the label, remove old pills
    const label = filter.querySelector(".blog-filter-label");
    filter.innerHTML = "";
    if (label) filter.appendChild(label);

    const allBtn = document.createElement("button");
    allBtn.className = "blog-filter-pill active";
    allBtn.dataset.tag = "all";
    allBtn.textContent = filterId.includes("zh") ? "全部" : "All";
    filter.appendChild(allBtn);

    allTags.forEach((tag) => {
      const btn = document.createElement("button");
      btn.className = "blog-filter-pill";
      btn.dataset.tag = tag;
      btn.textContent = tag;
      filter.appendChild(btn);
    });

    // Click handler
    filter.addEventListener("click", (e) => {
      if (!e.target.classList.contains("blog-filter-pill")) return;
      filter.querySelectorAll(".blog-filter-pill").forEach((b) => b.classList.remove("active"));
      e.target.classList.add("active");
      const activeTag = e.target.dataset.tag;
      const locale = filterId.includes("zh") ? "zh" : "en";
      const filtered = activeTag === "all" ? posts : posts.filter((p) => (p.tags || []).includes(activeTag));
      renderPosts(listEl, filtered, locale);
    });
  }

  function renderPosts(listEl, posts, locale) {
    if (posts.length === 0) {
      listEl.innerHTML =
        locale === "zh"
          ? '<li class="blog-empty">没有匹配的文章。</li>'
          : '<li class="blog-empty">No matching posts.</li>';
      return;
    }

    listEl.innerHTML = posts
      .map((p) => {
        const title = p.title || p.slug;
        const href = "/" + p.slug;
        // Brief description from content (first ~120 chars, strip HTML/markdown)
        const raw = (p.content || "").replace(/<[^>]*>/g, "").replace(/[#*`~\[\]()|]/g, "").trim();
        const desc = raw.length > 140 ? raw.slice(0, 140).replace(/\s\S*$/, "") + "…" : raw;
        const tagsHtml =
          (p.tags || []).length > 0
            ? (p.tags || [])
                .map((t) => '<span class="blog-item-tag">' + escapeHtml(t) + "</span>")
                .join("")
            : "";

        return (
          '<li class="blog-item">' +
          '<a class="blog-item-link" href="' +
          href +
          '">' +
          '<div class="blog-item-main">' +
          '<h3 class="blog-item-title">' +
          escapeHtml(title) +
          "</h3>" +
          '<p class="blog-item-desc">' +
          escapeHtml(desc) +
          "</p>" +
          "</div>" +
          '<div class="blog-item-tags">' +
          tagsHtml +
          "</div>" +
          "</a>" +
          "</li>"
        );
      })
      .join("");
  }

  function escapeHtml(str) {
    var div = document.createElement("div");
    div.appendChild(document.createTextNode(str));
    return div.innerHTML;
  }

  // Run on load, and on SPA navigation (Quartz nav event)
  init();
  document.addEventListener("nav", init);
})();
</script>
